Creación de Tablas

CREATE DATABASE `grupo3` DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;
USE `grupo3`;


create table  `grupo3`.`usuario`(
id_usuario varchar(9) not null primary key,
usuario varchar(50) not null unique,
contraseña varchar(15)not null,
email varchar(60)not null unique,
tipo varchar(50)not null,
unique (id_usuario ,email )
)ENGINE = INNODB;


CREATE table `grupo3`.`comentario`(
id_comentario varchar(9) not null primary key,
fecha_comentario date not null,
comentarios varchar(200)not null,
id_usuario varchar(9) not null,
foreign key (id_usuario) references usuario(id_usuario) on delete cascade,
unique (id_comentario)
)ENGINE = INNODB;


CREATE table `grupo3`.`entrada`(
id_entrada varchar(9) not null primary key,
fecha_entrada date not null,
entradas varchar(200)not null,
id_usuario varchar(9) not null,
foreign key (id_usuario) references usuario(id_usuario) on delete cascade,
unique (id_entrada)
)ENGINE = INNODB;
